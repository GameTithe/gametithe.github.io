#include <iostream>
#include <cstring>
#include <string>
#include <vector>
#include <map>
#include <queue>
#include <algorithm>
using namespace std;
 
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    vector<int> num = { 3, 2, 1, 	2, 	3, 	3	, 3, 	3, 	1, 	1, 	3, 	1, 	3, 	3, 	1, 	2, 	2, 	2, 	1, 	2, 	1	, 1	, 2, 	2, 	2, 	1 };
     
    string str;
    cin >> str;
     
    int total = 0;
    int len = str.length();
    for (int i = 0; i < len; i++)
    {
        total += num[str[i] - 'A'];
    } 

    total %= 2;

    if (total== 0 ) cout << "You're the winner?";
    else  cout << "I'm a winner!";
}
